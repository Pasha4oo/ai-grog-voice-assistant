import os

def open_paint():
    # Windows
    if os.name == 'nt':
        os.system('mspaint')
    # macOS
    elif os.name == 'posix':
        os.system('open -a Paintbrush')
    # Linux
    else:
        print("Sorry, I don't know how to open Paint on your operating system.")

if __name__ == "__main__":
    open_paint()
